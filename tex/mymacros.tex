\newcommand{\mcbpplevel}[1]{
\switch[#1=]
\case{1}
\case{2} \hspace{0.6cm} \coqhead{0.6cm}
\case{3} \coqhead{0.6cm} \coqhead{0.6cm}
\otherwise Please use \textbackslash mcbLEVEL
\endswitch
}
\newcommand{\mcbpptxtlevel}[1]{
\switch[#1=]
\case{1}
\case{2} $(\star)$
\case{3} $(\star\star)$
\otherwise Please use \textbackslash mcbLEVEL
\endswitch
}
